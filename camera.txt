#Pi_cam.py
from  piCamera import PiCamera
from time import sleep
int n=0
camera = PiCamera()
while(n<=10):
	camera.start_preview()
	sleep(5)
	camera.capture(“/home/pi/Desktop/image%s.jpg”,%n)
	camera.stop_preview()
	n=+1
